    aws-sigv4

    A dependency-free, test suite-compliant, AWS Signature Version 4 library in ES6

    Example

    var sigv4 = require('aws-sigv4');

sigv4.sign(
    secretAccessKey,
    requestDate.slice(0, 8),
    'us-east-1',
    'host',
    stringToSign
);
